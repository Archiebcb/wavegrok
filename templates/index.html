<body>

    <h1>WaveGrok Trading Beast</h1>

    <input type="text" id="symbol" placeholder="Symbol (e.g., BTC/USDT)">

    <input type="text" id="primary_tf" placeholder="Primary TF (e.g., 1h)">

    <input type="text" id="secondary_tf" placeholder="Secondary TF (e.g., 4h)">

    <input type="number" id="limit" placeholder="Candles (e.g., 100)">

    <button onclick="fetchData()">Fetch Data</button>

    <button onclick="analyze()">Analyze</button>

    <div id="message"></div>

    <img id="chart" src="" alt="Chart here">

    <pre id="results"></pre>

    <script>

        async function fetchData() {

            const symbol = document.getElementById('symbol').value;

            const timeframe = document.getElementById('primary_tf').value;

            const limit = document.getElementById('limit').value;

            const response = await fetch('/fetch', {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({ symbol, timeframe, limit })

            });

            const data = await response.json();

            document.getElementById('message').innerText = data.message;

            if (data.message.includes("Fetched")) {

                document.getElementById('chart').src = `/chart/${timeframe}?t=${Date.now()}`;

            }

        }

        async function analyze() {

            const symbol = document.getElementById('symbol').value;

            const primary_tf = document.getElementById('primary_tf').value;

            const secondary_tf = document.getElementById('secondary_tf').value;

            const response = await fetch('/analyze', {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({ symbol, primary_tf, secondary_tf })

            });

            const data = await response.json();

            document.getElementById('results').innerText = data.message;

        }

    </script>
